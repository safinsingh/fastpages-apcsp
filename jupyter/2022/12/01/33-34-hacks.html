num1 = 5
# num1 = 5

num2 = num1 * 3
# num2 = 15

num3 = num2 / num1 * (9 % 2) * 4
# num3 = 15 / 5 * (9 % 2) * 4
# num3 = 15 / 5 * 1 * 4
# num3 = 12

result = (num3 % num1 + num2) % num3 * 3 / 5
# result = (12 % 5 + 15) % 12 * 3 / 5
# result = (2 + 15) % 12 * 3 / 5
# result = 5 * 3 / 5
# result = 3

# check
print(result)
